Helper: จัดการ URL ตาม Service function getBaseUrl(endpoint) { if (endpoint.startsWith('/auth')) return API_CONFIG.AUTH_URL; if (endpoint.startsWith('/tasks')) return API_CONFIG.TASK_URL; if (endpoint.startsWith('/users')) return API_CONFIG.USER_URL; return API_CONFIG.AUTH_URL; } // 2. Smart Fetcher: ยิง API ไปแต่ละ Service และจัดการ Token/Error async function apiFetch(endpoint, options = {}) { const baseUrl = getBaseUrl(endpoint); const token = localStorage.getItem('jwt_token'); options.headers = { 'Content-Type': 'application/json', ...(token ? { 'Authorization': `Bearer ${token}` } : {}), ...options.headers }; try { const res = await fetch(`${baseUrl}/api${endpoint}`, options); // เก็บ Log (เพื่อให้ logs.html ทำงานได้) logApiCall(options.method || 'GET', `${baseUrl}/api${endpoint}`, res.status, 0); if (res.status === 401) { doLogout(); alert('Session หมดอายุ กรุณา Login ใหม่'); } return res; } catch (err) { console.error('API Error:', err); throw err; } } // 3. Auth Functions async function doLogin() { const email = document.getElementById('login-email').value; const password = document.getElementById('login-password').value; const res = await apiFetch('/auth/login', { method: 'POST', body: JSON.stringify({ email, password }) }); const data = await res.json(); if (res.ok) { localStorage.setItem('jwt_token', data.token); window.location.reload(); } else { showAuthMsg(data.error || 'Login ล้มเหลว', 'error'); } } function doLogout() { localStorage.removeItem('jwt_token'); window.location.href = 'index.html'; } // 4. Data Loading (จาก Microservices) async function initApp() { // โหลด Profile (User Service) const profRes = await apiFetch('/users/me', { method: 'GET' }); if (profRes.ok) { const user = await profRes.json(); document.getElementById('prof-name').textContent = user.display_name || user.username; document.getElementById('prof-email').textContent = user.email; document.getElementById('sidebar-name').textContent = user.username; document.getElementById('sidebar-role-badge').textContent = user.role; } // โหลด Tasks (Task Service) loadTasks(); } async function loadTasks() { const res = await apiFetch('/tasks', { method: 'GET' }); if (res.ok) { allTasks = await res.json(); renderTasks(); // ใช้ฟังก์ชันแสดงผลที่คุณมีอยู่แล้ว } } // 5. Admin Function async function loadAllUsers() { const res = await apiFetch('/users', { method: 'GET' }); if (res.status === 403) { document.getElementById('users-forbidden').classList.remove('hidden'); } else if (res.ok) { const users = await res.json(); renderUserTable(users); // ใช้ฟังก์ชันแสดงตารางที่คุณมี } } // เริ่มต้นระบบ window.onload = () => { if (localStorage.getItem('jwt_token')) { document.getElementById('auth-page').classList.add('hidden'); document.getElementById('app-page').classList.remove('hidden'); initApp(); } }; /* --- เก็บฟังก์ชัน logApiCall ที่คุณมีอยู่เดิมไว้ตรงนี้ --- */ function logApiCall(method, url, status, duration) { const logs = JSON.parse(localStorage.getItem('taskboard_api_log') || '[]'); logs.unshift({ time: new Date().toLocaleTimeString(), method, url, status }); localStorage.setItem('taskboard_api_log', JSON.stringify(logs.slice(0, 50))); }
